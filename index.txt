            modalBody.innerHTML = `

                    <img src="${bio.img}" class="w-full h-full object-cover rounded-full border-2 border-gold p-1" alt="${bio.name}">

                    <p class="text-gold-light text-xs uppercase tracking-widest mb-4 font-semibold">${bio.role}</p>
                    <p class="text-gray-300 leading-relaxed text-lg italic">"${bio.text}"</p>
                </div>
            `;
            modal.style.display = "flex";
            document.body.classList.add("modal-open");
        }

        // Función para cerrar la ventana modal
        function closeBio() {
            modal.style.display = "none";
            document.body.classList.remove("modal-open");
        }

        // Cerrar modal al hacer clic fuera del contenido
        window.onclick = function(event) {
            if (event.target == modal) closeBio();
        }
